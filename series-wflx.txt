server {
    server_name series.wflx.st;

    root /var/www/series.wflx.st;
    index index.html;

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/series.wflx.st/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/series.wflx.st/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

    # Reverse proxy for MP4/m3u8/ts streaming
    location ~* \.(mp4|m3u8|ts)$ {
        # proxy_pass https://series.shieldedstar.com;
        proxy_pass http://95.168.164.157;

        proxy_set_header Host series.shieldedstar.com;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_http_version 1.1;
        proxy_set_header Connection "";

        proxy_set_header Range $http_range;
        proxy_set_header If-Range $http_if_range;

        proxy_buffering off;
        proxy_request_buffering off;

        # Remove any existing CORS headers from upstream
        proxy_hide_header Access-Control-Allow-Origin;
        proxy_hide_header Access-Control-Allow-Methods;
        proxy_hide_header Access-Control-Allow-Headers;


        set $cors "";
        if ($http_origin ~* "^https?://([a-z0-9-]+\.)?wflx\.st$") {
          set $cors $http_origin;
        }

        add_header Access-Control-Allow-Origin $cors always;
        add_header Access-Control-Allow-Methods "GET, OPTIONS" always;
        add_header Access-Control-Allow-Headers "*" always;

        # CORS headers
        # add_header Access-Control-Allow-Origin *;
        # add_header Access-Control-Allow-Methods "GET, OPTIONS";
        # add_header Access-Control-Allow-Headers "*";
    }

}
