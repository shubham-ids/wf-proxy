server {
    server_name ms.watchflixad.com;

    root /var/www/ms.watchflixad.com;
    index index.html;

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/ms.watchflixad.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/ms.watchflixad.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot


    # Reverse proxy for MP4/m3u8/ts streaming
    location / {
        
        # proxy_pass https://madplay.site;
        # proxy_pass https://95.211.82.251;        
        # resolver 8.8.8.8 ipv6=off;
        # set $upstream "shieldedstar.com";
        # proxy_pass https://$upstream;


        # proxy_set_header Host shieldedstar.com;
        # proxy_set_header Host $host;
        
        resolver 8.8.8.8 1.1.1.1 ipv6=off;

        proxy_ssl_verify off;

        set $upstream madplay.site;
        proxy_pass https://$upstream;


        # proxy_pass https://madplay.site;

        proxy_ssl_server_name on;
        proxy_ssl_name madplay.site;

        proxy_set_header Host madplay.site;

        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_http_version 1.1;
        proxy_set_header Connection "";

        proxy_set_header Range $http_range;
        proxy_set_header If-Range $http_if_range;

        proxy_buffering off;
        proxy_request_buffering off;



        # Enable content substitution
        sub_filter_once off;  # Apply the substitution globally, not just the first occurrence
        sub_filter '//madplay.site' '//ms.watchflixad.com';  # Replace 'madplay.site' with 'ms.watchflixad.com'
        
        # Enable sub_filter for all content types (this works for HTML, m3u8, mp4, etc.)
        sub_filter_types *;



        # proxy_ssl_server_name on;
        # proxy_ssl_verify off;


        # CORS headers
        # add_header Access-Control-Allow-Origin *;
        #  add_header Access-Control-Allow-Methods "GET, OPTIONS";
        # add_header Access-Control-Allow-Headers "*";

        # CORS headers
        set $cors "";
        if ($http_origin ~* "\.wflx\.st$") {
            set $cors $http_origin;
        }
        if ($http_origin ~* "\.watchflixad\.com$") {
            set $cors $http_origin;
        }
        if ($http_origin ~* "\.watchflx\.tv$") {
            set $cors $http_origin;
        }
        if ($http_origin ~* "\.iflx\.tv$") {
            set $cors $http_origin;
        }

        # Apply CORS headers if the origin is allowed
        if ($cors != "") {
            add_header Access-Control-Allow-Origin $cors;
            add_header Access-Control-Allow-Methods "GET, OPTIONS";
            add_header Access-Control-Allow-Headers "*";
        }
        
        # Handle OPTIONS preflight request (CORS)
        if ($request_method = 'OPTIONS') {
            add_header Access-Control-Allow-Origin $cors;
            add_header Access-Control-Allow-Methods "GET, OPTIONS";
            add_header Access-Control-Allow-Headers "*";
            return 204;
        }



    }

}

    if ($host = ms.watchflixad.com) {
